Task 1 — Forest share per canton (terra)

library(terra)
library(dplyr)
library(ggplot2)
library(scales)
library(readr)
library(knitr)
library(tictoc)

tic.clearlog()

tlm_path <- "data/SWISSTLM3D_2026_LV95_LN02.gpkg"
bnd_path <- "data/swissBOUNDARIES3D_1_5_LV95_LN02.gpkg"
tic("Task 1: load cantons")
cantons <- vect(bnd_path, layer = "tlm_kantonsgebiet")
toc(log = TRUE)
Task 1: load cantons: 0.1 sec elapsed
tic("Task 1: load land cover")
surface <- vect(tlm_path, layer = "tlm_bb_bodenbedeckung")
toc(log = TRUE)
Task 1: load land cover: 15.9 sec elapsed
tic("Task 1: filter forest")
wald <- surface[surface$objektart == "Wald", ]
toc(log = TRUE)
Task 1: filter forest: 2.5 sec elapsed
tic("Task 1: prepare canton attributes")
cantons_small <- cantons[, c("name", "kantonsflaeche")]
toc(log = TRUE)
Task 1: prepare canton attributes: 0 sec elapsed
rm(surface, cantons)

tic("Task 1: intersect forest with cantons")
forest_per_canton <- terra::intersect(wald, cantons_small)
toc(log = TRUE)
Task 1: intersect forest with cantons: 134.63 sec elapsed
tic("Task 1: compute forest area")
forest_per_canton$forest_area_m2 <- expanse(forest_per_canton, unit = "m")
toc(log = TRUE)
Task 1: compute forest area: 37.9 sec elapsed
tic("Task 1: summarise by canton")
forest_by_canton <- aggregate(forest_area_m2 ~ name, data = forest_per_canton, FUN = sum)
toc(log = TRUE)
Task 1: summarise by canton: 0.24 sec elapsed
tic("Task 1: calculate forest share")
cantons_df <- as.data.frame(cantons_small) |>
  distinct(name, kantonsflaeche) |>
  mutate(kantonsflaeche_m2 = kantonsflaeche * 10000)

forest_terra <- cantons_df |>
  left_join(forest_by_canton, by = "name") |>
  mutate(
    forest_area_m2 = ifelse(is.na(forest_area_m2), 0, forest_area_m2),
    forest_share = forest_area_m2 / kantonsflaeche_m2
  ) |>
  arrange(desc(forest_share))
toc(log = TRUE)
Task 1: calculate forest share: 0.03 sec elapsed
forest_terra |>
  mutate(forest_share_pct = forest_share * 100) |>
  select(name, forest_share_pct) |>
  kable(digits = 2, col.names = c("Canton", "Forest share (%)"))

Task 2 — Forest share per canton (GDAL CLI)

dir.create("gdal_output", showWarnings = FALSE)
tictoc::tic("Task 2: extract forest with GDAL")
ogr2ogr -overwrite -f GPKG gdal_output/work.gpkg data/SWISSTLM3D_2026_LV95_LN02.gpkg tlm_bb_bodenbedeckung -dialect SQLITE -sql "SELECT * FROM tlm_bb_bodenbedeckung WHERE objektart = 'Wald'" -nln wald
tictoc::toc(log = TRUE)
Task 2: extract forest with GDAL: 0.14 sec elapsed
tictoc::tic("Task 2: extract cantons with GDAL")
ogr2ogr -update -append -f GPKG gdal_output/work.gpkg data/swissBOUNDARIES3D_1_5_LV95_LN02.gpkg tlm_kantonsgebiet -dialect SQLITE -sql "SELECT name, kantonsflaeche, geom FROM tlm_kantonsgebiet" -nln cantons
tictoc::toc(log = TRUE)
Task 2: extract cantons with GDAL: 0.16 sec elapsed
tictoc::tic("Task 2: intersect and summarise with GDAL")
ogr2ogr -overwrite -f CSV gdal_output/forest_summary.csv gdal_output/work.gpkg -dialect SQLITE -sql "SELECT c.name, c.kantonsflaeche, SUM(ST_Area(ST_Intersection(w.geom, c.geom))) AS forest_area_m2 FROM wald w JOIN cantons c ON ST_Intersects(w.geom, c.geom) GROUP BY c.name, c.kantonsflaeche"
tictoc::toc(log = TRUE)
